```typescript
updateAnnotation: (messageId: string, annotationId: string, patch: Partial<Annotation>) => {
  set((state) => ({
    messages: {
      ...state.messages,
      [messageId]: {
        ...state.messages[messageId]!,
        annotations: state.messages[messageId]!.annotations.map(a =>
```

```typescript
import type { ApiResponse, SearchResult } from '../../../shared/types';

// Converts SearchResult (backend shape) to SourceResult (frontend annotation shape)
export function toSourceResult(r: SearchResult): SourceResult {
  let domain = r.url;
  try { domain = new URL(r.url).hostname.replace(/^www\./, ''); } catch { /* invalid URL */ }
  return { title: r.title, url: r.url, domain, snippet: r.content };
}

export async function searchSources(
  params: { query: string },
  getToken: () => Promise<string | null>
): Promise<ApiResponse<{ results: SearchResult[]; citationNote: string }>> {
  return apiRequest<{ results: SearchResult[]; citationNote: string }>('/api/find-sources', {
    method: 'POST',
    body: JSON.stringify(params),
    getToken,
  });
}
```

```typescript
async generateCitationNote(results: SearchResult[], originalText: string): Promise<string> {
  const sourceList = results
    .map((r, i) => `${i + 1}. "${r.title}" (${r.url}): ${r.content.slice(0, 200)}`)
    .join('\n');

  const prompt = `You found these sources for the text: "${originalText.slice(0, 300)}"\n\nSources:\n${sourceList}\n\nWrite 1–2 sentences synthesizing what these sources say about the text. Be factual and concise. Return only the note, no preamble.`;

  let lastErr: Error | null = null;
  for (const model of FREE_TIER_MODELS) {
    try {
      const response = await ai.models.generateContent({
        model,
        contents: prompt,
        config: { systemInstruction: 'You are a research assistant. Return only the citation note.' },
      });
      return response.text ?? '';
    } catch (err) {
      lastErr = err instanceof Error ? err : new Error(String(err));
      if (!isRetryableError(err)) break;
      console.warn(`[GeminiProvider] ${model} unavailable for citation note, trying next...`);
    }
  }
  throw lastErr ?? new Error('All Gemini models unavailable');
}
```

**3. Extend backend/src/routes/find-sources.ts to call `generateCitationNote` and include result:**

```typescript
try {
  const results = await searchProvider.findSources(query.trim(), 3);

  // Generate Gemini synthesis note in parallel with nothing (sequential is fine — note needs results)

  if (results.length > 0) {

}
```
