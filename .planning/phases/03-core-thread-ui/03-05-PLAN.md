```tsx
export function BreadcrumbBar() {
  const { threads, activeThreadId, setActiveThread } = useSessionStore(
    (s) => ({ threads: s.threads, activeThreadId: s.activeThreadId, setActiveThread: s.setActiveThread })
  );
  const containerRef = useRef<HTMLDivElement>(null);
  const [showDropdown, setShowDropdown] = useState(false);

  const ancestry = activeThreadId ? selectThreadAncestry(threads, activeThreadId) : [];
  const currentThread = ancestry[ancestry.length - 1];
  const ancestors = ancestry.slice(0, -1);

  // Crumb label helper
  const getCrumbLabel = (thread: Thread) =>
    thread.messageIds.length === 0 ? 'New Chat' : thread.title;

  // Collapse logic: if ancestry.length <= 3, show all; if > 3, show first, "...", last two
  const shouldCollapse = ancestry.length > 3;
  const collapsedMiddle = shouldCollapse ? ancestry.slice(1, -2) : [];
  const visibleAncestors = shouldCollapse
    ? [ancestors[0], ...ancestors.slice(-2)]  // first + last 2 ancestors
    : ancestors;
  // When shouldCollapse: first ancestor, "..." button, then last 2 ancestors, then current

  // ResizeObserver: attach to containerRef — re-evaluate collapse on resize
  // (Since collapse logic is data-driven by ancestry.length, the ResizeObserver only
  //  needs to trigger a re-render if we want pixel-based collapse. For now,
  //  data-length-based collapse is sufficient. ResizeObserver can be a no-op re-render trigger.)

  // Dropdown close on outside click
  useEffect(() => {
    if (!showDropdown) return;
    const handler = (e: MouseEvent) => {
      if (containerRef.current && !containerRef.current.contains(e.target as Node)) {
        setShowDropdown(false);
      }
    };
    document.addEventListener('mousedown', handler);
    return () => document.removeEventListener('mousedown', handler);
  }, [showDropdown]);

  // Render crumbs
  // Chevron separator: <span className="mx-2 text-zinc-600">›</span>
}
```

Render logic for the crumb list:
1. If `!activeThreadId` return null
2. Map visible ancestors to clickable `<button>` elements
3. If shouldCollapse, insert the "..." button between first ancestor and last 2 ancestors
4. Append the current crumb as a non-clickable `<span>`
5. Dropdown: absolutely positioned div listing all `collapsedMiddle` + hidden crumbs as clickable buttons

Full component wraps in: `<div ref={containerRef} className="flex items-center h-full text-sm overflow-hidden relative">`

**Implement frontend/src/components/layout/SpineStrip.tsx:**

```tsx
import { useSessionStore } from '../../store/sessionStore';
import { selectCurrentThread } from '../../store/selectors';

export function SpineStrip() {
  const { threads, activeThreadId, setActiveThread } = useSessionStore(
    (s) => ({ threads: s.threads, activeThreadId: s.activeThreadId, setActiveThread: s.setActiveThread })
  );

  const currentThread = selectCurrentThread(threads, activeThreadId ?? '');

  if (!currentThread || currentThread.depth === 0 || !currentThread.parentThreadId) {
    return null;
  }

  const parentThread = threads[currentThread.parentThreadId];
  if (!parentThread) return null;

  return (
    <div
      className="w-7 flex-shrink-0 flex items-center justify-center cursor-pointer bg-zinc-900 hover:bg-zinc-800 transition-colors"
      style={{ borderLeft: `3px solid ${parentThread.accentColor}` }}
      onClick={() => setActiveThread(currentThread.parentThreadId!)}
      title={`Back to: ${parentThread.title}`}
    >
      <span
        className="text-xs text-zinc-400 overflow-hidden"
        style={{
          writingMode: 'vertical-rl',
          textOrientation: 'mixed',
          transform: 'rotate(180deg)',
          maxHeight: '120px',
          display: 'block',
          overflow: 'hidden',
          whiteSpace: 'nowrap',
        }}
      >
        {parentThread.title}
      </span>
    </div>
  );
}
```

Note: In AppShell, SpineStrip is rendered BEFORE the main content div (to the left). The `<div className="flex h-screen ...">` in AppShell renders as: `[SpineStrip | main flex-col]`. When SpineStrip returns null (depth === 0), the main content takes full width.
