```typescript
function handleGoDeeper(anchorText: string, paragraphId: string) {
  if (!activeThread || !activeThreadId) return;
  const accentColor = getNextAccentColor(activeThread);
  const title = anchorText.split(' ').slice(0, 6).join(' ');
  // Find the last AI message to use as parentMessageId
  const lastAiMsg = [...orderedMessages].reverse().find(m => m.role === 'assistant');
  const newThreadId = createThread({
    parentThreadId: activeThreadId,
    anchorText,
    parentMessageId: lastAiMsg?.id ?? null,
    title,
    accentColor,
  });
  if (lastAiMsg) {
    addChildLead(lastAiMsg.id, {
      threadId: newThreadId,
      paragraphIndex: Number(paragraphId),
      anchorText,
      messageCount: 0,
    });
  }
  clearBubble();
  setActiveThread(newThreadId);
}
```

The underline logic requires knowing the accentColor per childLead. The `ChildLead` type has `threadId` — but the thread accentColor is in the store. To keep MarkdownRenderer pure (no store access), pass a pre-computed map from ThreadView/MessageBlock:

In `MessageBlock.tsx`, read the store to build an underline map:
```typescript
const threads = useSessionStore(s => s.threads);

// Build map: paragraphIndex -> accentColor (from this message's childLeads)
const underlineMap: Record<number, string> = {};
for (const lead of message.childLeads) {
  const threadColor = threads[lead.threadId]?.accentColor;
  if (threadColor) underlineMap[lead.paragraphIndex] = threadColor;
}
```

Pass `underlineMap` down to `MarkdownRenderer` as a new prop:
```typescript
// MessageBlock renders:
<MarkdownRenderer content={message.content} underlineMap={underlineMap} />
```

Update `MarkdownRenderer.tsx` to accept `underlineMap?: Record<number, string>` prop and use it in a custom `p` component:
```typescript
components={{
  p({ children, ...props }) {
    const paragraphId = (props as Record<string, unknown>)['data-paragraph-id'];
    const color = typeof paragraphId === 'string'
      ? underlineMap?.[Number(paragraphId)]
      : undefined;
    return (
      <p
        {...props}
        style={color ? { textDecoration: 'underline', textDecorationColor: color } : undefined}
      >
        {children}
      </p>
    );
  },
}}
```

NOTE: The `data-paragraph-id` attribute from the rehype plugin will be in the props passed to the custom `p` component.
