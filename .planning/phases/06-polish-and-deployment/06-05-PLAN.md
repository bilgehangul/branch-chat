**1. Implement `frontend/e2e/go-deeper.spec.ts`** — Flow 3: branch creation.

The go-deeper flow requires: (a) send a message to get an AI response, (b) select text in the response, (c) action bubble appears, (d) click "Go Deeper". The action bubble appears at the end of mouse selection — in Playwright, we simulate this with `page.mouse` operations.

```typescript
import { test, expect } from '@playwright/test';
import { readFileSync } from 'fs';
import { join } from 'path';

const SSE_FIXTURE = readFileSync(join(__dirname, 'fixtures/chat-stream.txt'), 'utf-8');

test.describe('go deeper', () => {
  test('user can branch into a child thread via Go Deeper', async ({ page }) => {
    await page.route('**/api/chat', route => {
      route.fulfill({
        status: 200,
        contentType: 'text/event-stream',
        body: SSE_FIXTURE,
      });
    });

    await page.goto('/');

    // Send a message and wait for AI response
    const textarea = page.locator('textarea[placeholder="Ask anything..."]');
    await textarea.fill('Tell me about branching');
    await page.getByRole('button', { name: 'Send' }).click();

    // Wait for AI response text to appear
    const aiResponse = page.locator('text=DeepDive').first();
    await expect(aiResponse).toBeVisible({ timeout: 10000 });

    // Select text in the AI response using mouse drag
    // Get the bounding box of the response text to select
    const responseEl = page.locator('.prose, [data-message-role="assistant"]').first();
    await responseEl.waitFor({ state: 'visible', timeout: 5000 });

    const box = await responseEl.boundingBox();
    if (box) {
      // Drag to select text within the response
      await page.mouse.move(box.x + 10, box.y + 10);
      await page.mouse.down();
      await page.mouse.move(box.x + 150, box.y + 10);
      await page.mouse.up();
    }

    // Action bubble should appear
    const goDeeperBtn = page.getByRole('button', { name: 'Go Deeper' });
    await expect(goDeeperBtn).toBeVisible({ timeout: 3000 });

    // Click Go Deeper
    await goDeeperBtn.click();

    // A new child thread should open — the breadcrumb should show 2 levels
    // Or: the "Ask anything to begin" state changes / context card appears
    await expect(
      page.locator('text=Ask anything to begin').or(page.locator('[data-testid="context-card"]'))
    ).toBeVisible({ timeout: 5000 });
  });
});
```

import { test, expect } from '@playwright/test';
import { readFileSync } from 'fs';
import { join } from 'path';

const SSE_FIXTURE = readFileSync(join(__dirname, 'fixtures/chat-stream.txt'), 'utf-8');

test.describe('navigation', () => {
  test('breadcrumb shows thread path and root is clickable', async ({ page }) => {
    await page.route('**/api/chat', route => {
      route.fulfill({
        status: 200,
        contentType: 'text/event-stream',
        body: SSE_FIXTURE,
      });
    });

    await page.goto('/');

    // Initial breadcrumb shows root (before any messages: may be empty or "New Chat")
    // After sending a message, the thread title is set to first 6 words
    await page.locator('textarea[placeholder="Ask anything..."]').fill('Navigation breadcrumb test');
    await page.getByRole('button', { name: 'Send' }).click();
    await expect(page.locator('text=DeepDive').first()).toBeVisible({ timeout: 10000 });

    // Breadcrumb bar should be visible
    const header = page.locator('header');
    await expect(header).toBeVisible();

    // Create a child thread to test breadcrumb depth
    const responseEl = page.locator('.prose, [data-message-role="assistant"]').first();
    const box = await responseEl.boundingBox();
    if (box) {
      await page.mouse.move(box.x + 10, box.y + 10);
      await page.mouse.down();
      await page.mouse.move(box.x + 150, box.y + 10);
      await page.mouse.up();
    }

    const goDeeperBtn = page.getByRole('button', { name: 'Go Deeper' });
    await expect(goDeeperBtn).toBeVisible({ timeout: 3000 });
    await goDeeperBtn.click();

    // Now in child thread — breadcrumb should show 2+ levels
    // The header still shows navigation elements
    await expect(header).toBeVisible();
  });

  test('Go Deeper is disabled and shows tooltip at max depth (depth 4)', async ({ page }) => {
    await page.route('**/api/chat', route => {
      route.fulfill({
        status: 200,
        contentType: 'text/event-stream',
        body: SSE_FIXTURE,
      });
    });

    await page.goto('/');

    // Build a 4-deep thread tree by creating 4 child threads
    // Each iteration: send a message, select text, click Go Deeper
    async function sendAndGoDeeper() {
      await page.locator('textarea[placeholder="Ask anything..."]').fill('depth test message');
      await page.getByRole('button', { name: 'Send' }).click();
      await expect(page.locator('text=DeepDive').first()).toBeVisible({ timeout: 10000 });

      const responseEl = page.locator('.prose, [data-message-role="assistant"]').first();
      const box = await responseEl.boundingBox();
      if (!box) return false;
      await page.mouse.move(box.x + 10, box.y + 10);
      await page.mouse.down();
      await page.mouse.move(box.x + 150, box.y + 10);
      await page.mouse.up();

      const goDeeperBtn = page.getByRole('button', { name: 'Go Deeper' });
      // If button exists but is not disabled, we're not at max depth yet
      try {
        await expect(goDeeperBtn).toBeVisible({ timeout: 2000 });
        const isDisabled = await goDeeperBtn.isDisabled();
        if (isDisabled) return false; // at max depth
        await goDeeperBtn.click();
        return true;
      } catch {
        return false;
      }
    }

    // Navigate to depth 4 (requires 4 Go Deeper clicks from root at depth 0)
    let canGoDeeper = true;
    for (let i = 0; i < 5 && canGoDeeper; i++) {
      canGoDeeper = await sendAndGoDeeper();
    }

    // At depth 4, the Go Deeper button should be disabled
    // Select text to show the action bubble
    await page.locator('textarea[placeholder="Ask anything..."]').fill('depth limit check');
    await page.getByRole('button', { name: 'Send' }).click();
    await expect(page.locator('text=DeepDive').first()).toBeVisible({ timeout: 10000 });

    const responseEl = page.locator('.prose, [data-message-role="assistant"]').first();
    const box = await responseEl.boundingBox();
    if (box) {
      await page.mouse.move(box.x + 10, box.y + 10);
      await page.mouse.down();
      await page.mouse.move(box.x + 150, box.y + 10);
      await page.mouse.up();
    }

    const goDeeperBtn = page.getByRole('button', { name: 'Go Deeper' });
    await expect(goDeeperBtn).toBeVisible({ timeout: 3000 });
    await expect(goDeeperBtn).toBeDisabled();
  });
});
